#include<stdio.h>
int main()
{
        int num,temp,rev=0;
        printf("Enter the num to be reversed\n");
        scanf("%d",&num);
        while(num)
        {
                temp=num%10;
                rev=rev*10+temp;
                num/=10;
        }
        printf("The Reversed number is %d\n",rev);--

}
